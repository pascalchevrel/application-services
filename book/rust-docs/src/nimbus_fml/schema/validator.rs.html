37    pub(crate) fn validate_object_def(&self, object_def: &ObjectDef) -> Result<()> {
38        let obj_nm = &object_def.name;
39        for prop in &object_def.props {
40            let prop_nm = &prop.name;
41
42            // Check the types exist for this property.
43            let path = format!("objects/{obj_nm}/{prop_nm}");
44            self.validate_type_ref(&path, &prop.typ)?;
45        }
46
47        Ok(())
48    }
49
50    pub(crate) fn validate_feature_def(&self, feature_def: &FeatureDef) -> Result<()> {
51        let feat_nm = &feature_def.name;
52        let mut string_aliases: HashSet<_> = Default::default();
53
54        for prop in &feature_def.props {
55            let prop_nm = &prop.name;
56            let prop_t = &prop.typ;
57
58            let path = format!("features/{feat_nm}/{prop_nm}");
59
60            // Check the types exist for this property.
61            self.validate_type_ref(&path, prop_t)?;
62
63            // Check pref support for this type.
64            if prop.pref_key.is_some() && !prop.typ.supports_prefs() {
65                return Err(FMLError::ValidationError(
66                    path,
67                    "Pref keys can only be used with Boolean, String, Int and Text variables"
68                        .to_string(),
69                ));
70            }
71
72            // Check string-alias definition.
73            if let Some(sa) = &prop.string_alias {
74                // Check that the string-alias has only been defined once in this feature.
75                if !string_aliases.insert(sa) {
76                    return Err(FMLError::ValidationError(
77                        path,
78                        format!("The string-alias {sa} should only be declared once per feature"),
79                    ));
80                }
81
82                // Check that the string-alias is actually used in this property type.
83                let types = prop_t.all_types();
84                if !types.contains(sa) {
85                    return Err(FMLError::ValidationError(
86                        path,
87                        format!(
88                            "The string-alias {sa} must be part of the {} type declaration",
89                            prop_nm
90                        ),
91                    ));
92                }
93            }
94        }
95
96        // Now check that that there is a path from this feature to any objects using the
97        // string-aliases defined in this feature.
98        let types = feature_def.all_types();
99        self.validate_string_alias_declarations(
100            &format!("features/{feat_nm}"),
101            feat_nm,
102            &types,
103            &string_aliases,
104        )?;
105
106        Ok(())
107    }
108
109    fn validate_string_alias_declarations(
110        &self,
111        path: &str,
112        feature: &str,
113        types: &HashSet<TypeRef>,
114        string_aliases: &HashSet<&TypeRef>,
115    ) -> Result<()> {
116        let unaccounted: Vec<_> = types
117            .iter()
118            .filter(|t| matches!(t, TypeRef::StringAlias(_)))
119            .filter(|t| !string_aliases.contains(t))
120            .collect();
121
122        if !unaccounted.is_empty() {
123            let t = unaccounted.first().unwrap();
124            return Err(FMLError::ValidationError(
125                path.to_string(),
126                format!("A string-alias {t} is used by– but has not been defined in– the {feature} feature"),
127            ));
128        }
129        for t in types {
130            if let TypeRef::Object(nm) = t {
131                if let Some(obj) = self.get_object(nm) {
132                    let types = obj.all_types();
133                    self.validate_string_alias_declarations(
134                        &format!("objects/{nm}"),
135                        feature,
136                        &types,
137                        string_aliases,
138                    )?;
139                }
140            }
141        }
142        Ok(())
143    }
144
145    fn validate_type_ref(&self, path: &str, type_ref: &TypeRef) -> Result<()> {
146        match type_ref {
147            TypeRef::Enum(name) => {
148                if !self.enum_defs.contains_key(name) {
149                    return Err(FMLError::ValidationError(
150                        path.to_string(),
151                        format!("Found enum reference with name: {name}, but no definition"),
152                    ));
153                }
154            }
155            TypeRef::Object(name) => {
156                if !self.object_defs.contains_key(name) {
157                    return Err(FMLError::ValidationError(
158                        path.to_string(),
159                        format!("Found object reference with name: {name}, but no definition"),
160                    ));
161                }
162            }
163            TypeRef::EnumMap(key_type, value_type) => match key_type.as_ref() {
164                TypeRef::Enum(_) | TypeRef::String | TypeRef::StringAlias(_) => {
165                    self.validate_type_ref(path, key_type)?;
166                    self.validate_type_ref(path, value_type)?;
167                }
168                _ => {
169                    return Err(FMLError::ValidationError(
170                        path.to_string(),
171                        format!(
172                            "Map key must be a String, string-alias or enum, found: {key_type:?}",
173                        ),
174                    ))
175                }
176            },
177            TypeRef::List(list_type) => self.validate_type_ref(path, list_type)?,
178            TypeRef::StringMap(value_type) => self.validate_type_ref(path, value_type)?,
179            TypeRef::Option(option_type) => {
180                if let TypeRef::Option(_) = option_type.as_ref() {
181                    return Err(FMLError::ValidationError(
182                        path.to_string(),
183                        "Found nested optional types".into(),
184                    ));
185                } else {
186                    self.validate_type_ref(path, option_type)?
187                }
188            }
189            _ => (),
190        };
191        Ok(())
192    }
193}

386    fn with_objects(objects: &[ObjectDef]) -> BTreeMap<String, ObjectDef> {
387        let mut obj_defs: BTreeMap<_, _> = Default::default();
388        for o in objects {
389            obj_defs.insert(o.name(), o.clone());
390        }
391        obj_defs
392    }
393
394    fn with_feature(props: &[PropDef]) -> FeatureDef {
395        FeatureDef::new("test-feature", "", props.into(), false)
396    }
397
398    #[test]
399    fn test_validate_feature_schema() -> Result<()> {
400        let name = TypeRef::StringAlias("PersonName".to_string());
401        let all_names = {
402            let t = TypeRef::List(Box::new(name.clone()));
403            let v = json!(["Alice", "Bonnie", "Charlie", "Denise", "Elise", "Frankie"]);
404            PropDef::with_string_alias("all-names", &t, &v, &name)
405        };
406
407        let all_names2 = {
408            let t = TypeRef::List(Box::new(name.clone()));
409            let v = json!(["Alice", "Bonnie"]);
410            PropDef::with_string_alias("all-names-duplicate", &t, &v, &name)
411        };
412
413        let enums = Default::default();
414        let objects = Default::default();
415        let validator = SchemaValidator::new(&enums, &objects);
416
417        // -> Verify that only one property per feature can define the same string-alias.
418        let fm = with_feature(&[all_names.clone(), all_names2.clone()]);
419        assert!(validator.validate_feature_def(&fm).is_err());
420
421        let newest_member = {
422            let t = &name;
423            let v = json!("Alice"); // it doesn't matter for this test what the value is.
424            PropDef::new("newest-member", t, &v)
425        };
426
427        // -> Verify that a property in a feature can validate against the a string-alias
428        // -> in the same feature.
429        // { all-names: ["Alice"], newest-member: "Alice" }
430        let fm = with_feature(&[all_names.clone(), newest_member.clone()]);
431        validator.validate_feature_def(&fm)?;
432
433        // { newest-member: "Alice" }
434        // We have a reference to a team mate, but no definitions.
435        // Should error out.
436        let fm = with_feature(&[newest_member.clone()]);
437        assert!(validator.validate_feature_def(&fm).is_err());
438
439        // -> Validate a property in a nested object can validate against a string-alias
440        // -> in a feature that uses the object.
441        let team_def = ObjectDef::new("Team", &[newest_member.clone()]);
442        let team = {
443            let t = TypeRef::Object("Team".to_string());
444            let v = json!({ "newest-member": "Alice" });
445
446            PropDef::new("team", &t, &v)
447        };
448
449        // { all-names: ["Alice"], team: { newest-member: "Alice" } }
450        let fm = with_feature(&[all_names.clone(), team.clone()]);
451        let objs = with_objects(&[team_def.clone()]);
452        let validator = SchemaValidator::new(&enums, &objs);
453        validator.validate_feature_def(&fm)?;
454
455        // { team: { newest-member: "Alice" } }
456        let fm = with_feature(&[team.clone()]);
457        let objs = with_objects(&[team_def.clone()]);
458        let validator = SchemaValidator::new(&enums, &objs);
459        assert!(validator.validate_feature_def(&fm).is_err());
460
461        // -> Validate a property in a deeply nested object can validate against a string-alias
462        // -> in a feature that uses the object.
463
464        let match_def = ObjectDef::new("Match", &[team.clone()]);
465        let match_ = {
466            let t = TypeRef::Object("Match".to_string());
467            let v = json!({ "team": { "newest-member": "Alice" }});
468
469            PropDef::new("match", &t, &v)
470        };
471
472        // { all-names: ["Alice"], match: { team: { newest-member: "Alice" }} }
473        let fm = with_feature(&[all_names.clone(), match_.clone()]);
474        let objs = with_objects(&[team_def.clone(), match_def.clone()]);
475        let validator = SchemaValidator::new(&enums, &objs);
476        validator.validate_feature_def(&fm)?;
477
478        // { match: {team: { newest-member: "Alice" }} }
479        let fm = with_feature(&[match_.clone()]);
480        let validator = SchemaValidator::new(&enums, &objs);
481        assert!(validator.validate_feature_def(&fm).is_err());
482
483        Ok(())
484    }
