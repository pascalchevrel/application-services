44    fn reconcile(
45        &self,
46        records: Vec<SyncLoginData>,
47        server_now: ServerTimestamp,
48        telem: &mut telemetry::EngineIncoming,
49    ) -> Result<UpdatePlan> {
50        let mut plan = UpdatePlan::default();
51
52        for mut record in records {
53            self.scope.err_if_interrupted()?;
54            debug!("Processing remote change {}", record.guid());
55            let upstream = if let Some(inbound) = record.inbound.take() {
56                inbound
57            } else {
58                debug!("Processing inbound deletion (always prefer)");
59                plan.plan_delete(record.guid.clone());
60                continue;
61            };
62            let upstream_time = record.inbound_ts;
63            match (record.mirror.take(), record.local.take()) {
64                (Some(mirror), Some(local)) => {
65                    debug!("  Conflict between remote and local, Resolving with 3WM");
66                    plan.plan_three_way_merge(
67                        local,
68                        mirror,
69                        upstream,
70                        upstream_time,
71                        server_now,
72                        self.store.encdec.as_ref(),
73                    )?;
74                    telem.reconciled(1);
75                }
76                (Some(_mirror), None) => {
77                    debug!("  Forwarding mirror to remote");
78                    plan.plan_mirror_update(upstream, upstream_time);
79                    telem.applied(1);
80                }
81                (None, Some(local)) => {
82                    debug!("  Conflicting record without shared parent,  Resolving with 2WM");
83                    plan.plan_two_way_merge(local, (upstream, upstream_time));
84                    telem.reconciled(1);
85                }
86                (None, None) => {
87                    if let Some(dupe) = self.find_dupe_login(&upstream.login)? {
88                        debug!(
89                            "  Incoming record {} was is a dupe of local record {}",
90                            upstream.guid(),
91                            dupe.guid()
92                        );
93                        let local_modified = UNIX_EPOCH
94                            + Duration::from_millis(dupe.meta.time_password_changed as u64);
95                        let local = LocalLogin::Alive {
96                            login: dupe,
97                            local_modified,
98                        };
99                        plan.plan_two_way_merge(local, (upstream, upstream_time));
100                    } else {
101                        debug!("  No dupe found, inserting into mirror");
102                        plan.plan_mirror_insert(upstream, upstream_time, false);
103                    }
104                    telem.applied(1);
105                }
106            }
107        }
108        Ok(plan)
109    }

111    fn execute_plan(&self, plan: UpdatePlan) -> Result<()> {
112        // Because rusqlite want a mutable reference to create a transaction
113        // (as a way to save us from ourselves), we side-step that by creating
114        // it manually.
115        let db = self.store.db.lock();
116        let tx = db.unchecked_transaction()?;
117        plan.execute(&tx, &self.scope)?;
118        tx.commit()?;
119        Ok(())
120    }

123    // TODO: Might be better taking a fn instead of returning all of it... But that func will likely

125    fn fetch_login_data(
126        &self,
127        records: Vec<IncomingBso>,
128        telem: &mut telemetry::EngineIncoming,
129    ) -> Result<Vec<SyncLoginData>> {
130        let mut sync_data = Vec::with_capacity(records.len());
131        {
132            let mut seen_ids: HashSet<Guid> = HashSet::with_capacity(records.len());
133            for incoming in records.into_iter() {
134                let id = incoming.envelope.id.clone();
135                match SyncLoginData::from_bso(incoming, self.store.encdec.as_ref()) {
136                    Ok(v) => sync_data.push(v),
137                    Err(e) => {
138                        match e {
139                            // This is a known error with Desktop logins (see #5233), just log it
140                            // rather than reporting to sentry
141                            Error::InvalidLogin(InvalidLogin::IllegalOrigin) => {
142                                warn!("logins-deserialize-error: {e}");
143                            }
144                            // For all other errors, report them to Sentry
145                            _ => {
146                                report_error!(
147                                    "logins-deserialize-error",
148                                    "Failed to deserialize record {:?}: {e}",
149                                    id
150                                );
151                            }
152                        };
153                        // Ideally we'd track new_failed, but it's unclear how
154                        // much value it has.
155                        telem.failed(1);
156                    }
157                }
158                seen_ids.insert(id);
159            }
160        }
161        self.scope.err_if_interrupted()?;
162
163        sql_support::each_chunk(
164            &sync_data
165                .iter()
166                .map(|s| s.guid.as_str().to_string())
167                .collect::<Vec<String>>(),
168            |chunk, offset| -> Result<()> {
169                // pairs the bound parameter for the guid with an integer index.
170                let values_with_idx = sql_support::repeat_display(chunk.len(), ",", |i, f| {
171                    write!(f, "({},?)", i + offset)
172                });
173                let query = format!(
174                    "WITH to_fetch(guid_idx, fetch_guid) AS (VALUES {vals})
175                     SELECT
176                         {common_cols},
177                         is_overridden,
178                         server_modified,
179                         NULL as local_modified,
180                         NULL as is_deleted,
181                         NULL as sync_status,
182                         1 as is_mirror,
183                         to_fetch.guid_idx as guid_idx
184                     FROM loginsM
185                     JOIN to_fetch
186                         ON loginsM.guid = to_fetch.fetch_guid
187
188                     UNION ALL
189
190                     SELECT
191                         {common_cols},
192                         NULL as is_overridden,
193                         NULL as server_modified,
194                         local_modified,
195                         is_deleted,
196                         sync_status,
197                         0 as is_mirror,
198                         to_fetch.guid_idx as guid_idx
199                     FROM loginsL
200                     JOIN to_fetch
201                         ON loginsL.guid = to_fetch.fetch_guid",
202                    // give each VALUES item 2 entries, an index and the parameter.
203                    vals = values_with_idx,
204                    common_cols = schema::COMMON_COLS,
205                );
206
207                let db = &self.store.db.lock();
208                let mut stmt = db.prepare(&query)?;
209
210                let rows = stmt.query_and_then(rusqlite::params_from_iter(chunk), |row| {
211                    let guid_idx_i = row.get::<_, i64>("guid_idx")?;
212                    // Hitting this means our math is wrong...
213                    assert!(guid_idx_i >= 0);
214
215                    let guid_idx = guid_idx_i as usize;
216                    let is_mirror: bool = row.get("is_mirror")?;
217                    if is_mirror {
218                        sync_data[guid_idx].set_mirror(MirrorLogin::from_row(row)?)?;
219                    } else {
220                        sync_data[guid_idx].set_local(LocalLogin::from_row(row)?)?;
221                    }
222                    self.scope.err_if_interrupted()?;
223                    Ok(())
224                })?;
225                // `rows` is an Iterator<Item = Result<()>>, so we need to collect to handle the errors.
226                rows.collect::<Result<()>>()?;
227                Ok(())
228            },
229        )?;
230        Ok(sync_data)
231    }

233    fn fetch_outgoing(&self) -> Result<Vec<OutgoingBso>> {
234        // Taken from iOS. Arbitrarily large, so that clients that want to
235        // process deletions first can; for us it doesn't matter.
236        const TOMBSTONE_SORTINDEX: i32 = 5_000_000;
237        const DEFAULT_SORTINDEX: i32 = 1;
238        let db = self.store.db.lock();
239        let mut stmt = db.prepare_cached(&format!(
240            "SELECT L.*, M.enc_unknown_fields
241             FROM loginsL L LEFT JOIN loginsM M ON L.guid = M.guid
242             WHERE sync_status IS NOT {synced}",
243            synced = SyncStatus::Synced as u8
244        ))?;
245        let bsos = stmt.query_and_then([], |row| {
246            self.scope.err_if_interrupted()?;
247            Ok(if row.get::<_, bool>("is_deleted")? {
248                let envelope = OutgoingEnvelope {
249                    id: row.get::<_, String>("guid")?.into(),
250                    sortindex: Some(TOMBSTONE_SORTINDEX),
251                    ..Default::default()
252                };
253                OutgoingBso::new_tombstone(envelope)
254            } else {
255                let unknown = row.get::<_, Option<String>>("enc_unknown_fields")?;
256                let mut bso =
257                    EncryptedLogin::from_row(row)?.into_bso(self.store.encdec.as_ref(), unknown)?;
258                bso.envelope.sortindex = Some(DEFAULT_SORTINDEX);
259                bso
260            })
261        })?;
262        bsos.collect::<Result<_>>()
263    }

265    fn do_apply_incoming(
266        &self,
267        inbound: Vec<IncomingBso>,
268        timestamp: ServerTimestamp,
269        telem: &mut telemetry::Engine,
270    ) -> Result<Vec<OutgoingBso>> {
271        let mut incoming_telemetry = telemetry::EngineIncoming::new();
272        let data = self.fetch_login_data(inbound, &mut incoming_telemetry)?;
273        let plan = {
274            let result = self.reconcile(data, timestamp, &mut incoming_telemetry);
275            telem.incoming(incoming_telemetry);
276            result
277        }?;
278        self.execute_plan(plan)?;
279        self.fetch_outgoing()
280    }

307    fn mark_as_synchronized(&self, guids: &[&str], ts: ServerTimestamp) -> Result<()> {
308        let db = self.store.db.lock();
309        let tx = db.unchecked_transaction()?;
310        sql_support::each_chunk(guids, |chunk, _| -> Result<()> {
311            db.execute(
312                &format!(
313                    "DELETE FROM loginsM WHERE guid IN ({vars})",
314                    vars = sql_support::repeat_sql_vars(chunk.len())
315                ),
316                rusqlite::params_from_iter(chunk),
317            )?;
318            self.scope.err_if_interrupted()?;
319
320            db.execute(
321                &format!(
322                    "INSERT OR IGNORE INTO loginsM (
323                         {common_cols}, is_overridden, server_modified
324                     )
325                     SELECT {common_cols}, 0, {modified_ms_i64}
326                     FROM loginsL
327                     WHERE is_deleted = 0 AND guid IN ({vars})",
328                    common_cols = schema::COMMON_COLS,
329                    modified_ms_i64 = ts.as_millis(),
330                    vars = sql_support::repeat_sql_vars(chunk.len())
331                ),
332                rusqlite::params_from_iter(chunk),
333            )?;
334            self.scope.err_if_interrupted()?;
335
336            db.execute(
337                &format!(
338                    "DELETE FROM loginsL WHERE guid IN ({vars})",
339                    vars = sql_support::repeat_sql_vars(chunk.len())
340                ),
341                rusqlite::params_from_iter(chunk),
342            )?;
343            self.scope.err_if_interrupted()?;
344            Ok(())
345        })?;
346        self.set_last_sync(&db, ts)?;
347        tx.commit()?;
348        Ok(())
349    }

351    // This exists here as a public function so the store can call it. Ideally
352    // the store would not do that :) Then it can go back into the sync trait
353    // and return an anyhow::Result
354    pub fn do_reset(&self, assoc: &EngineSyncAssociation) -> Result<()> {
355        info!("Executing reset on password engine!");
356        let db = self.store.db.lock();
357        let tx = db.unchecked_transaction()?;
358        db.execute_all(&[
359            &CLONE_ENTIRE_MIRROR_SQL,
360            "DELETE FROM loginsM",
361            &format!("UPDATE loginsL SET sync_status = {}", SyncStatus::New as u8),
362        ])?;
363        self.set_last_sync(&db, ServerTimestamp(0))?;
364        match assoc {
365            EngineSyncAssociation::Disconnected => {
366                db.delete_meta(schema::GLOBAL_SYNCID_META_KEY)?;
367                db.delete_meta(schema::COLLECTION_SYNCID_META_KEY)?;
368            }
369            EngineSyncAssociation::Connected(ids) => {
370                db.put_meta(schema::GLOBAL_SYNCID_META_KEY, &ids.global)?;
371                db.put_meta(schema::COLLECTION_SYNCID_META_KEY, &ids.coll)?;
372            }
373        };
374        db.delete_meta(schema::GLOBAL_STATE_META_KEY)?;
375        tx.commit()?;
376        Ok(())
377    }

379    // It would be nice if this were a batch-ish api (e.g. takes a slice of records and finds dupes
380    // for each one if they exist)... I can't think of how to write that query, though.
381    // This is subtly different from dupe handling by the main API and maybe
382    // could be consolidated, but for now it remains sync specific.
383    pub(crate) fn find_dupe_login(&self, l: &EncryptedLogin) -> Result<Option<EncryptedLogin>> {
384        let form_submit_host_port = l
385            .fields
386            .form_action_origin
387            .as_ref()
388            .and_then(|s| util::url_host_port(s));
389        let enc_fields = l.decrypt_fields(self.store.encdec.as_ref())?;
390        let args = named_params! {
391            ":origin": l.fields.origin,
392            ":http_realm": l.fields.http_realm,
393            ":form_submit": form_submit_host_port,
394        };
395        let mut query = format!(
396            "SELECT {common}
397             FROM loginsL
398             WHERE origin IS :origin
399               AND httpRealm IS :http_realm",
400            common = schema::COMMON_COLS,
401        );
402        if form_submit_host_port.is_some() {
403            // Stolen from iOS
404            query += " AND (formActionOrigin = '' OR (instr(formActionOrigin, :form_submit) > 0))";
405        } else {
406            query += " AND formActionOrigin IS :form_submit"
407        }
408        let db = self.store.db.lock();
409        let mut stmt = db.prepare_cached(&query)?;
410        for login in stmt
411            .query_and_then(args, EncryptedLogin::from_row)?
412            .collect::<Result<Vec<EncryptedLogin>>>()?
413        {
414            let this_enc_fields = login.decrypt_fields(self.store.encdec.as_ref())?;
415            if enc_fields.username == this_enc_fields.username {
416                return Ok(Some(login));
417            }
418        }
419        Ok(None)
420    }

423impl SyncEngine for LoginsSyncEngine {
424    fn collection_name(&self) -> std::borrow::Cow<'static, str> {
425        "passwords".into()
426    }
427
428    fn stage_incoming(
429        &self,
430        mut inbound: Vec<IncomingBso>,
431        _telem: &mut telemetry::Engine,
432    ) -> anyhow::Result<()> {
433        // We don't have cross-item dependencies like bookmarks does, so we can
434        // just apply now instead of "staging"
435        self.staged.borrow_mut().append(&mut inbound);
436        Ok(())
437    }
438
439    fn apply(
440        &self,
441        timestamp: ServerTimestamp,
442        telem: &mut telemetry::Engine,
443    ) -> anyhow::Result<Vec<OutgoingBso>> {
444        let inbound = (*self.staged.borrow_mut()).drain(..).collect();
445        Ok(self.do_apply_incoming(inbound, timestamp, telem)?)
446    }
447
448    fn set_uploaded(&self, new_timestamp: ServerTimestamp, ids: Vec<Guid>) -> anyhow::Result<()> {
449        Ok(self.mark_as_synchronized(
450            &ids.iter().map(Guid::as_str).collect::<Vec<_>>(),
451            new_timestamp,
452        )?)
453    }
454
455    fn get_collection_request(
456        &self,
457        server_timestamp: ServerTimestamp,
458    ) -> anyhow::Result<Option<CollectionRequest>> {
459        let db = self.store.db.lock();
460        let since = self.get_last_sync(&db)?.unwrap_or_default();
461        Ok(if since == server_timestamp {
462            None
463        } else {
464            Some(
465                CollectionRequest::new("passwords".into())
466                    .full()
467                    .newer_than(since),
468            )
469        })
470    }
471
472    fn get_sync_assoc(&self) -> anyhow::Result<EngineSyncAssociation> {
473        let db = self.store.db.lock();
474        let global = db.get_meta(schema::GLOBAL_SYNCID_META_KEY)?;
475        let coll = db.get_meta(schema::COLLECTION_SYNCID_META_KEY)?;
476        Ok(if let (Some(global), Some(coll)) = (global, coll) {
477            EngineSyncAssociation::Connected(CollSyncIds { global, coll })
478        } else {
479            EngineSyncAssociation::Disconnected
480        })
481    }
482
483    fn reset(&self, assoc: &EngineSyncAssociation) -> anyhow::Result<()> {
484        self.do_reset(assoc)?;
485        Ok(())
486    }

837                .store

849            let changeset = engine.fetch_outgoing().unwrap();

852        }

856        let engine = LoginsSyncEngine::new(Arc::new(store)).unwrap();

924    fn do_test_incoming_with_local_unmirrored_tombstone(local_newer: bool) {
925        ensure_initialized();
926        fn apply_incoming_payload(engine: &LoginsSyncEngine, payload: serde_json::Value) {
927            let bso = IncomingBso::from_test_content(payload);
928            let mut telem = sync15::telemetry::Engine::new(engine.collection_name());
929            engine.stage_incoming(vec![bso], &mut telem).unwrap();
930            engine
931                .apply(ServerTimestamp::from_millis(0), &mut telem)
932                .unwrap();
933        }
934
935        // The test itself...
936        let (local_timestamp, remote_timestamp) = if local_newer { (123, 0) } else { (0, 123) };
937
938        let store = LoginStore::new_in_memory(TEST_ENCDEC.clone()).unwrap();
939        let engine = LoginsSyncEngine::new(Arc::new(store)).unwrap();
940
941        // apply an incoming record - will be in the mirror.
942        apply_incoming_payload(
943            &engine,
944            serde_json::json!({
945                "id": "dummy_000001",
946                "formSubmitURL": "https://www.example.com/submit",
947                "hostname": "https://www.example.com",
948                "username": "test",
949                "password": "test",
950                "timePasswordChanged": local_timestamp,
951                "unknown1": "?",
952                "unknown2": {"sub": "object"},
953            }),
954        );
955
956        // Reset the engine - this wipes the mirror.
957        engine.reset(&EngineSyncAssociation::Disconnected).unwrap();
958        // But the local record does still exist.
959        assert!(engine
960            .store
961            .get("dummy_000001")
962            .expect("should work")
963            .is_some());
964
965        // Delete the local record.
966        engine.store.delete("dummy_000001").unwrap();
967        assert!(engine
968            .store
969            .get("dummy_000001")
970            .expect("should work")
971            .is_none());
972
973        // double-check our test preconditions - should now have 1 in LoginsL and 0 in LoginsM
974        assert_eq!(count(&engine, "LoginsL"), 1);
975        assert_eq!(count(&engine, "LoginsM"), 0);
976
977        // Now we assume we've been reconnected to sync and have an incoming change for the record.
978        apply_incoming_payload(
979            &engine,
980            serde_json::json!({
981                "id": "dummy_000001",
982                "formSubmitURL": "https://www.example.com/submit",
983                "hostname": "https://www.example.com",
984                "username": "test",
985                "password": "test2",
986                "timePasswordChanged": remote_timestamp,
987                "unknown1": "?",
988                "unknown2": {"sub": "object"},
989            }),
990        );
991
992        // Desktop semantics here are that a local tombstone is treated as though it doesn't exist at all.
993        // ie, the remote record should be taken whether it is newer or older than the tombstone.
994        assert!(engine
995            .store
996            .get("dummy_000001")
997            .expect("should work")
998            .is_some());
999        // and there should never be an outgoing record.
1000        // XXX - but there is! But this is exceedingly rare, we
1001        // should fix it :)
1002        // assert_eq!(engine.fetch_outgoing().unwrap().len(), 0);
1003
1004        // should now be no records in loginsL and 1 in loginsM
1005        assert_eq!(count(&engine, "LoginsL"), 0);
1006        assert_eq!(count(&engine, "LoginsM"), 1);
1007    }
