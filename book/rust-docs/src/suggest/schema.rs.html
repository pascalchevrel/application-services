13/// The current database schema version.
14///
15/// For any changes to the schema [`SQL`], please make sure to:
16///
17///  1. Bump this version.
18///  2. Add a migration from the old version to the new version in
19///     [`SuggestConnectionInitializer::upgrade_from`].
20///     a. If suggestions should be re-ingested after the migration, call `clear_database()` inside
21///     the migration.
22///  3. If the migration adds any tables, delete their their rows in
23///     `clear_database()` by adding their names to `conditional_tables`, unless
24///     they are cleared via a deletion trigger or there's some other good
25///     reason not to do so.
26pub const VERSION: u32 = 41;

251/// Initializes an SQLite connection to the Suggest database, performing
252/// migrations as needed.
253#[derive(Default)]
254pub struct SuggestConnectionInitializer<'a> {
255    extensions_to_load: &'a [Sqlite3Extension],
256}
257
258impl<'a> SuggestConnectionInitializer<'a> {
259    pub fn new(extensions_to_load: &'a [Sqlite3Extension]) -> Self {
260        Self { extensions_to_load }
261    }
262
263    pub fn load_extensions(&self, conn: &Connection) -> open_database::Result<()> {
264        // Safety: this relies on the extensions we're loading to operate correctly, for the
265        // entry point to be correct, etc.
266        unsafe {
267            let _guard = rusqlite::LoadExtensionGuard::new(conn)?;
268            for ext in self.extensions_to_load {
269                conn.load_extension(&ext.library, ext.entry_point.as_deref())?;
270            }
271        }
272        Ok(())
273    }
274
275    fn create_custom_functions(&self, conn: &Connection) -> open_database::Result<()> {
276        conn.create_collation("geonames_collate", geonames_collate)?;
277        Ok(())
278    }
279}

285    fn prepare(&self, conn: &Connection, _db_empty: bool) -> open_database::Result<()> {
286        self.load_extensions(conn)?;
287        sql_support::setup_sqlite_defaults(conn)?;
288        conn.execute("PRAGMA foreign_keys = ON", ())?;
289        sql_support::debug_tools::define_debug_functions(conn)?;
290        self.create_custom_functions(conn)?;
291        Ok(())
292    }
293
294    fn init(&self, db: &Transaction<'_>) -> open_database::Result<()> {
295        db.execute_batch(SQL)?;
296        Ok(())
297    }

741                    CREATE TABLE keywords_metrics(
742                        record_id TEXT NOT NULL PRIMARY KEY,
743                        record_type TEXT NOT NULL,
744                        max_len INTEGER NOT NULL,
745                        max_word_count INTEGER NOT NULL
746                    ) WITHOUT ROWID;
747                    CREATE INDEX keywords_metrics_record_type ON keywords_metrics(record_type);
748                    "#,
749                )?;
750                Ok(())
751            }
752            40 => {
753                // This migration makes changes to geonames.
754                clear_database(tx)?;
755                tx.execute_batch(
756                    r#"
757                    DROP INDEX geonames_alternates_geoname_id;
758                    DROP INDEX geonames_alternates_name;
759                    DROP TABLE geonames_alternates;
760
761                    CREATE TABLE geonames_alternates(
762                        id INTEGER PRIMARY KEY,
763                        record_id TEXT NOT NULL,
764                        geoname_id INTEGER NOT NULL,
765                        language TEXT,
766                        name TEXT NOT NULL COLLATE geonames_collate,
767                        is_preferred INTEGER,
768                        is_short INTEGER
769                    );
770                    CREATE INDEX geonames_alternates_geoname_id_language
771                        ON geonames_alternates(geoname_id, language);
772                    CREATE INDEX geonames_alternates_name
773                        ON geonames_alternates(name);
774                    "#,
775                )?;
776                Ok(())
777            }
778            _ => Err(open_database::Error::IncompatibleVersion(version)),
779        }
780    }
781}

783/// Clears the database, removing all suggestions, icons, and metadata.
784pub fn clear_database(db: &Connection) -> rusqlite::Result<()> {
785    db.execute_batch(
786        "
787        DELETE FROM meta;
788        DELETE FROM keywords;
789        DELETE FROM full_keywords;
790        DELETE FROM prefix_keywords;
791        DELETE FROM suggestions;
792        DELETE FROM icons;
793        DELETE FROM yelp_subjects;
794        DELETE FROM yelp_modifiers;
795        DELETE FROM yelp_custom_details;
796        ",
797    )?;
798    let conditional_tables = [
799        "fakespot_fts",
800        "geonames",
801        "geonames_metrics",
802        "ingested_records",
803        "keywords_metrics",
804    ];
805    for t in conditional_tables {
806        let table_exists = db.exists("SELECT 1 FROM sqlite_master WHERE name = ?", [t])?;
807        if table_exists {
808            db.execute(&format!("DELETE FROM {t}"), ())?;
809        }
810    }
811    Ok(())
812}
